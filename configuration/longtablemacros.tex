% macros used in results table developed with Torben Westphalen.

\providecommand\itemcell[1]{%
  %\mbox{% uncomment for debug 
  \begin{minipage}[t]{\linewidth}%
      \begin{itemize}[leftmargin=3.5ex]%
        #1
      \end{itemize}%
  \end{minipage}\vspace{.4\baselineskip}%
  %}% uncomment
}

\providecommand\headcell[1]{%
  \begin{minipage}[c]{\linewidth}
    \vspace{.2\baselineskip}%
    \raggedright%
    \textbf{#1}
  \end{minipage}\vspace{.4\baselineskip}%
}

\providecommand\firstcell[1]{%
  \cellcolor{Gray}%
  \begin{minipage}[c]{\linewidth}%
    \vspace{.2\baselineskip}%
    \raggedright%
    \textbf{#1}
  \end{minipage}\vspace{.4\baselineskip}%
}

\providecommand\textcell[1]{%
  \fbox{%
    \begin{minipage}[t]{\linewidth}%
      \raggedright%
      #1
  \end{minipage}}\vspace{.4\baselineskip}%
}
